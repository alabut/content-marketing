Navigation

Example:

How It Works · About · Explore · Pricing · Sign Up Log In


HTML:

(notice that the "navigation" div block is inside the "banner" div)

                


        <div class="banner blog">
            <div class="navigation">
                <div class="container">
                    <nav class="navbar" role="navigation">
                        <!-- Brand and toggle get grouped for better mobile display -->
                        <div class="navbar-header">
                            <button type="button" class="navbar-toggle" data-toggle="collapse" data-target=".navbar-ex1-collapse">
                                <span class="sr-only">Toggle navigation</span>
                                <span class="icon-bar"></span>
                                <span class="icon-bar"></span>
                                <span class="icon-bar"></span>
                            </button>
                            <a class="navbar-brand" href="#">
                                <img src="images/logo-white-small.png" alt="Zapier logo">
                            </a>
                        </div><!-- /navbar-header -->

                        <!-- Collect the nav links, forms, and other content for toggling -->
                        <div class="collapse navbar-collapse navbar-ex1-collapse">

                            <p class="navbar-text pull-right">
                                <a href="https://zapier.com/how-it-works/" class="navbar-link">How It Works</a>
                                <strong>&middot;</strong>
                                <a href="https://zapier.com/about/" class="navbar-link">About</a>
                                <strong>&middot;</strong>
                                <a href="https://zapier.com/app/explore" class="navbar-link">Explore</a>
                                <strong>&middot;</strong>
                                <a href="https://zapier.com/app/pricing" class="navbar-link">Pricing</a>
                                <strong>&middot;</strong>
                                <a href="https://zapier.com/app/signup" class="navbar-link">Sign Up</a>
                                <a href="https://zapier.com/app/login" class="btn btn-default navbar-btn">Log In</a>
                            </p>

                        </div><!-- /.navbar-collapse -->
                    </nav>

                </div><!-- /container -->
            </div><!-- /navigation -->
            <div class="container">
                <h1>How KISSmetrics Increased Webinar Signup Rates by 1,000%</h1>
            </div><!-- /container -->
        </div><!-- /banner -->
